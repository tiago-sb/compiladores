program merge_sort;
var
  a: array[1..10] of integer;
  b: array[1..10] of integer;
  i, j, k, inicio, fim, meio: integer;
begin
  a[1] := 5; a[2] := 1; a[3] := 4; a[4] := 2; a[5] := 8;
  a[6] := 3; a[7] := 6; a[8] := 9; a[9] := 7; a[10] := 0;

  inicio := 1;
  meio := 5;
  fim := 10;
  
  for i := inicio to fim do
    b[i] := a[i];
    
  i := inicio;
  j := meio + 1;
  k := inicio;

  while (i <= meio) and (j <= fim) do
  begin
    if b[i] <= b[j] then
    begin
      a[k] := b[i];
      i := i + 1;
    end
    else
    begin
      a[k] := b[j];
      j := j + 1;
    end;
    k := k + 1;
  end;
  
  while i <= meio do
  begin
    a[k] := b[i];
    k := k + 1;
    i := i + 1;
  end;
end.
