program bubble_sort;
var
  a: array [1..10] of integer;
  i, j, temp: integer;
begin
  // Inicializa o array para demonstracao
  a[1] := 5; a[2] := 1; a[3] := 4; a[4] := 2; a[5] := 8;
  a[6] := 3; a[7] := 6; a[8] := 9; a[9] := 7; a[10] := 0;

  for i := 1 to 9 do
    for j := 1 to 10 - i do
      if a[j] > a[j+1] then
      begin
        temp := a[j];
        a[j] := a[j+1];
        a[j+1] := temp;
      end;
end.
