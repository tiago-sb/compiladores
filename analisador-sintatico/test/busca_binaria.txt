program busca_binaria;
var
  a: array [1..10] of integer;
  inicio, fim, meio, valor, encontrado: integer;
begin
  // Array ordenado para busca binaria
  a[1] := 0; a[2] := 1; a[3] := 2; a[4] := 3; a[5] := 4;
  a[6] := 5; a[7] := 6; a[8] := 7; a[9] := 8; a[10] := 9;

  valor := 6;
  inicio := 1;
  fim := 10;
  encontrado := -1;

  while inicio <= fim do
  begin
    meio := (inicio + fim) div 2;
    if a[meio] = valor then
    begin
      encontrado := meio;
      fim := inicio - 1; // Encerra o loop
    end
    else if a[meio] < valor then
      inicio := meio + 1
    else
      fim := meio - 1;
  end;
end.
