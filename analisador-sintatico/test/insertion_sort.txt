program insertion_sort;
var
  a: array [1..10] of integer;
  i, j, chave: integer;
begin
  // Inicializa o array para demonstracao
  a[1] := 5; a[2] := 1; a[3] := 4; a[4] := 2; a[5] := 8;
  a[6] := 3; a[7] := 6; a[8] := 9; a[9] := 7; a[10] := 0;

  for i := 2 to 10 do
  begin
    chave := a[i];
    j := i - 1;

    while (j > 0) and (a[j] > chave) do
    begin
      a[j+1] := a[j];
      j := j - 1;
    end;
    a[j+1] := chave;
  end;
end.
