program quick_sort;
var
  a: array[1..10] of integer;
  i, j, pivo, temp, esquerda, direita, chave: integer;
begin

  a[1] := 5; a[2] := 1; a[3] := 4; a[4] := 2; a[5] := 8;
  a[6] := 3; a[7] := 6; a[8] := 9; a[9] := 7; a[10] := 0;


  esquerda := 1;
  direita := 10;
  
  if esquerda < direita then
  begin
    chave := a[esquerda];
    pivo := esquerda;

    for i := esquerda + 1 to direita do
      if a[i] < chave then
      begin
        pivo := pivo + 1;
        temp := a[i];
        a[i] := a[pivo];
        a[pivo] := temp;
      end;
    
    temp := a[esquerda];
    a[esquerda] := a[pivo];
    a[pivo] := temp;
  end;
end.
